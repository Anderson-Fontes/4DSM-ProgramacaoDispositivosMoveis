-- 1. EXTENSÃO PARA CRIPTOGRAFIA (Necessária para criar o Master via SQL)
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- 2. TABELA DE USUÁRIOS (Autenticação Central)
CREATE TABLE usuarios (
    id SERIAL PRIMARY KEY,
    email VARCHAR(100) UNIQUE NOT NULL,
    senha_hash VARCHAR(255) NOT NULL,
    perfil VARCHAR(20) CHECK (perfil IN ('aluno', 'professor', 'diretor', 'master')) NOT NULL
);

-- 3. TABELA DE ALUNOS
CREATE TABLE alunos (
    id SERIAL PRIMARY KEY,
    usuario_id INT REFERENCES usuarios(id) ON DELETE CASCADE,
    nome VARCHAR(100) NOT NULL,
    matricula VARCHAR(20) UNIQUE NOT NULL,
    curso VARCHAR(50),
    telefone VARCHAR(15),
    cep VARCHAR(10),
    endereco VARCHAR(150),
    cidade VARCHAR(50),
    estado VARCHAR(2)
);

-- 4. TABELA DE PROFESSORES
CREATE TABLE professores (
    id SERIAL PRIMARY KEY,
    usuario_id INT REFERENCES usuarios(id) ON DELETE CASCADE,
    nome VARCHAR(100) NOT NULL,
    titulacao VARCHAR(50),
    area VARCHAR(50)
);

-- 5. TABELA DE DISCIPLINAS (Com Sala e Horário para a Grade Semanal)
CREATE TABLE disciplinas (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    carga_horaria INT,
    professor_id INT REFERENCES professores(id),
    curso VARCHAR(50),
    semestre INT,
    dia_semana INT CHECK (dia_semana BETWEEN 1 AND 7), -- 1: Segunda, 5: Sexta
    horario_inicio TIME,
    sala VARCHAR(20)
);

-- 6. TABELA DE ATIVIDADES (Avaliações com Pesos %)
CREATE TABLE atividades (
    id SERIAL PRIMARY KEY,
    disciplina_id INT REFERENCES disciplinas(id) ON DELETE CASCADE,
    nome VARCHAR(100) NOT NULL,
    peso_percentual NUMERIC(5,2) NOT NULL -- Ex: 40.0 para 40%
);

-- 7. TABELA DE NOTAS
CREATE TABLE notas_atividades (
    id SERIAL PRIMARY KEY,
    atividade_id INT REFERENCES atividades(id) ON DELETE CASCADE,
    aluno_id INT REFERENCES alunos(id) ON DELETE CASCADE,
    nota NUMERIC(5,2) NOT NULL,
    CONSTRAINT unica_nota_aluno_atv UNIQUE(atividade_id, aluno_id)
);

-- 8. TABELA DE CHAMADAS (Frequência)
CREATE TABLE chamadas (
    id SERIAL PRIMARY KEY,
    disciplina_id INT REFERENCES disciplinas(id) ON DELETE CASCADE,
    aluno_id INT REFERENCES alunos(id) ON DELETE CASCADE,
    data_aula DATE DEFAULT CURRENT_DATE,
    presente BOOLEAN NOT NULL
);

-- 9. TABELA DE SOLICITAÇÕES (Entrada/Saída de Matérias)
CREATE TABLE solicitacoes (
    id SERIAL PRIMARY KEY,
    aluno_id INT REFERENCES alunos(id) ON DELETE CASCADE,
    disciplina_id INT REFERENCES disciplinas(id) ON DELETE CASCADE,
    tipo VARCHAR(20) CHECK (tipo IN ('entrada', 'saida')),
    status VARCHAR(20) DEFAULT 'pendente' CHECK (status IN ('pendente', 'aprovado', 'rejeitado')),
    data_solicitacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
